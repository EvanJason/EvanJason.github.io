function quickSort(arr) {
  // йҖ’еҪ’еҮәеҸЈ
  if (arr.length <= 1) {
    return arr;
  }
  
  // йҖүжӢ©еҹәеҮҶеҖјпјҲдёӯй—ҙдҪҚзҪ®пјү
  const pivotIndex = Math.floor(arr.length / 2);
  const pivot = arr.splice(pivotIndex, 1)[0];
  
  // е®ҡд№үе·ҰеҸіж•°з»„
  const left = [];
  const right = [];
  
  // еҲҶеҢәж“ҚдҪң
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] < pivot) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }
  
  // йҖ’еҪ’жҺ’еәҸе№¶еҗҲе№¶
  return quickSort(left).concat([pivot], quickSort(right));
}

// жөӢиҜ•
const arr = [5, 2, 8, 1, 9, 3];
console.log(quickSort(arr)); // [1, 2, 3, 5, 8, 9]

function debounce(fn, delay = 300) {
  let timer = null;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// дҪҝз”Ё
button.addEventListener('click', debounce(handleClick, 300));

ж–№жЎҲдәҢпјҡиҠӮжөҒпјҲThrottleпјүвҖӢ

function throttle(fn, interval = 300) {
  let lastTime = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}

// дҪҝз”Ё
button.addEventListener('click', throttle(handleClick, 300));

function escapeHtml(str) {
  const map = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#039;'
  };
  return str.replace(/[&<>"']/g, m => map[m]);
}
